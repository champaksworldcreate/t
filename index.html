# StudentDB Pickle with Undo/Redo
import pickle, os, copy

FILE="studentdb.pkl"
def load():
    if os.path.exists(FILE):
        with open(FILE,"rb") as f: return pickle.load(f)
    return {}

def save(db):
    with open(FILE,"wb") as f: pickle.dump(db,f)

def main():
    db=load(); undo=[]; redo=[]
    while True:
        print("1 Add 2 View 3 Undo 4 Redo 5 Save&Exit")
        c=input("Choice: ")
        if c=="1":
            undo.append(copy.deepcopy(db)); redo.clear()
            r=int(input("Roll: ")); n=input("Name: ")
            db[r]=n
        elif c=="2": print(db)
        elif c=="3" and undo:
            redo.append(copy.deepcopy(db)); db=undo.pop()
        elif c=="4" and redo:
            undo.append(copy.deepcopy(db)); db=redo.pop()
        elif c=="5":
            save(db); break

if __name__=="__main__": main()

# StudentDB JSON with Undo/Redo
import json, os, copy

FILE="studentdb.json"
def load():
    if os.path.exists(FILE):
        with open(FILE,"r") as f: return json.load(f)
    return {}

def save(db):
    with open(FILE,"w") as f: json.dump(db,f,indent=2)

def main():
    db=load(); undo=[]; redo=[]
    while True:
        print("1 Add 2 View 3 Undo 4 Redo 5 Save&Exit")

            redo.append(copy.deepcopy(db)); db=undo.pop()
        elif c=="4" and redo:
            undo.append(copy.deepcopy(db)); db=redo.pop()

            save(db); break

if __name__=="__main__": main()
